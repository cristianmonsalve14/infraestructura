export function catalogOptionLabel(
  options: CatalogOption[],
  value?: string | null,
): string {
  if (!value) return "—";
  const match = options.find(
    (option) =>
      option.value === value ||
      option.label === value ||
      option.value.toLowerCase() === value.toLowerCase(),
  );
  return match?.label ?? value;
}
